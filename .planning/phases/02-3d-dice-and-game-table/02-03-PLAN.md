1. **Create src/stores/diceStore.ts** -- Zustand store for dice state:
```typescript
import { create } from 'zustand';
import type { DieValue } from '../engine/types.js';
import { DICE_COUNT } from '../engine/constants.js';

export type DicePhase = 'idle' | 'rolling' | 'settled';

export interface DieState {
  value: DieValue | null;
  selected: boolean;
  canScore: boolean;
}

interface DiceStore {
  phase: DicePhase;
  dice: DieState[];
  rollCount: number;

  // Actions
  startRoll: () => void;
  setSettled: (values: DieValue[]) => void;
  selectDie: (index: number) => void;
  deselectDie: (index: number) => void;
  setCanScore: (indices: number[]) => void;
  reset: () => void;
}

const createInitialDice = (): DieState[] =>
  Array.from({ length: DICE_COUNT }, () => ({
    value: null,
    selected: false,
    canScore: false,
  }));

export const useDiceStore = create<DiceStore>((set) => ({
  phase: 'idle',
  dice: createInitialDice(),
  rollCount: 0,

  startRoll: () => set((s) => ({
    phase: 'rolling',
    rollCount: s.rollCount + 1,
    dice: s.dice.map((d) => ({ ...d, value: null, selected: false, canScore: false })),
  })),

  setSettled: (values) => set((s) => ({
    phase: 'settled',
    dice: s.dice.map((d, i) => ({ ...d, value: values[i] ?? null })),
  })),

  selectDie: (index) => set((s) => ({
    dice: s.dice.map((d, i) => i === index ? { ...d, selected: true } : d),
  })),

  deselectDie: (index) => set((s) => ({
    dice: s.dice.map((d, i) => i === index ? { ...d, selected: false } : d),
  })),

  setCanScore: (indices) => set((s) => ({
    dice: s.dice.map((d, i) => ({ ...d, canScore: indices.includes(i) })),
  })),

  reset: () => set({
    phase: 'idle',
    dice: createInitialDice(),
    rollCount: 0,
  }),
}));
```

   - Store refs to all 6 RigidBody instances using `useRef<RapierRigidBody[]>([])`
   - **Roll function** (`triggerRoll`):

     - Wake up all bodies: `body.wakeUp()`
     - Call `useDiceStore.getState().startRoll()` to set phase to 'rolling'
   - **Settle detection** in `useFrame`:
     - Only check when store phase is 'rolling'
     - Check if ALL 6 rigid bodies are sleeping: `body.isSleeping()` for each
     - When all sleeping, read each body's quaternion via `body.rotation()`, pass to `getUpFace()`, collect DieValue array
     - Call `useDiceStore.getState().setSettled(values)`
     - Add a debounce: require all bodies sleeping for 10 consecutive frames (~167ms at 60fps) before reading, to avoid premature reads if a die briefly stops then gets bumped
   - **Temporary roll trigger**: Add a click handler on the table felt mesh that calls `triggerRoll()` (this will be replaced by UI buttons in Phase 3, but needed now for testing)
